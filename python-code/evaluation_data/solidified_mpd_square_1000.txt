def heuristic_2(self, board, current_depth, max_depth):
    board_outcome = board.get_outcome()
    if current_depth == max_depth:
        if board_outcome != 0:
            if board_outcome == self.player:
                return 1
            else:
                return -1
        else:
            return 0
    if board_outcome != 0:
        if board_outcome == self.player:
            return max_depth - current_depth + 1
        else:
            return -(max_depth - current_depth + 1)
    else:
        score = 0
        for successor in self.get_successors(board):
            score += self.heuristic_2(successor[0], current_depth + 1, max_depth)
        return score
